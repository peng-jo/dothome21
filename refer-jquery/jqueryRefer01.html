형제 선택자 $("div + p") - div 태그의 인접한 형태 p 태그 중 첫 번째 태그만 선택합니다.
형제들 선택자 $("div ~ p") - div 태그의 인접한 형제 p 태그 중 모든 형제 태그를 선택합니다.

01. 기본 선택자

  • $("li")
  • $(".class")
  • $("#id")
  • $(".class, #id")
  • $("*")
리셋 클릭하면 클래스(class)를 선택합니다. 클릭하면 아이디(#id)를 선택합니다. 클릭하면 기본(li) 태그를 선택합니다. 클릭하면 여러가지(#id, .class) 태그를 선택합니다. 클릭하면 전체(*) 태그를 선택합니다. 
{
      $('.reset1').click(function(){
        $('.view1 *').removeClass("red");
      });
      $(".btn1-1").click(function(){
        $('.view1 .class').addClass("red");
      });
      $(".btn1-2").click(function(){
        $('.view1 #id').addClass("red");
      });
      $(".btn1-3").click(function(){
        $('.view1 li').addClass("red");
      });
      $(".btn1-4").click(function(){
        $('.view1 .class, .view1 #id').addClass("red");
      });
      $(".btn1-5").click(function(){
        $('.view1 li *').addClass("red");
      });
}

01. 기본 선택자- javascript

  • $("li")
  • $(".class")
  • $("#id")
  • $(".class, #id")
  • $("*")
리셋 클릭하면 클래스(class)를 선택합니다. 클릭하면 아이디(#id)를 선택합니다. 클릭하면 기본(li) 태그를 선택합니다. 
{
  const resetList = document.querySelectorAll(".jview1 *");
    for(let i = 0 ; i < resetList.length ; i ++){
      resetList[i].classList.remove("red");
    }
  });

  //버튼을 클릭했을 때 실행해라
  document.querySelector(".jbtn1-1").addEventListener("click",function(){
    document.querySelector(".jview1 .j1_class").classList.add("red");
  });

  document.querySelector(".jbtn1-2").addEventListener("click",function(){
    document.querySelector("#j1_id").classList.add("red");
  });

  document.querySelector(".jbtn1-3").addEventListener("click",function(){
    const liList = document.querySelectorAll(".jview1 li");
    // for문
    for(let i = 0 ; i < liList.length ; i ++){
      liList[i].classList.add("red");
    }
  });
}

02. 계층 선택자

div p p p p p p p p
리셋 클릭하면 $("div p") 선택합니다. 클릭하면 $("div > p") 선택합니다. 클릭하면 $("div + p") 선택합니다. 클릭하면 $("div ~ p") 선택합니다. 
{
  $('.reset2').click(function(){
    $('.s2 span').removeClass('blue');
  });
  $('.btn2-1').click(function(){
    $('.s2 span').addClass('blue');
  });
  $('.btn2-2').click(function(){
    $('.s2 > span').addClass('blue');
  });
  $('.btn2-3').click(function(){
    $('.s2 .select  + span').addClass('blue');
  });
  $('.btn2-4').click(function(){
    $('.s2 .select ~ span').addClass('blue');
  });
}
