19. 콜백 함수 : 동기/비동기 

{
    //01
    document.write("*** 19. 콜백 함수 : 동기/비동기 *** ");

    function funcA(){
        document.write("funcA가 실행되었습니다.");

    }

    function funcB(){
        document.write("funcB가 실행되었습니다.");
    }
    funcA();
    funcB();

    //02
    function funcC(){
        setTimeout(function(){
            console.log("funC가 실행되었습니다.");
        },1000);

    }

    function funcD(){
        console.log("funcD가 실행되었습니다.");
    }
    funcC();
    funcD();

    //03
    function funcE(callback){
        setTimeout(function(){
            console.log("funE가 실행되었습니다.");
            callback()
        },1000);

    }
    function funcF(){
        console.log("funF가 실행되었습니다.");
    }
    funcE(funcF);
}

25. 함수 유형 : 함수와 객체(함수)를 이용한 형태 

{
    document.write(" *** 25. 함수 유형 : 함수와 객체(함수)를 이용한 형태 *** ");
    
    const info = {
        num1: 1,
        name1: "조범수",
        job1: "웹 퍼블리셔",
        num2: 2,
        name2: "조범수",
        job2: "프론트앤드 개발자",
        result1: function(){
            document.write(`${this.num1}. 내이름은 ${this.name1}이며 직업은 ${this.job1} 입니다.`);
        },
        result2: function(){
            document.write(`${this.num2}. 내이름은 ${this.name2}이며 직업은 ${this.job2} 입니다.`);
        }
    }
    info.result1();
    info.result2();
}

26. 함수 유형 : 객체 생성자 함수 

{
    document.write(" *** 26. 함수 유형 : 객체 생성자 함수 *** ");

    function Func(num, name, job){
        this.num  = num;
        this.name = name;
        this.job  = job;
        this.result = function(){
            document.write(`${this.num}. 내이름은 ${this.name}이며 직업은 ${this.job} 입니다.`);
        }
    }    

    const info1 = new Func(1, "조범수", "웹퍼블리셔"); //인스턴스 생성
    const info2 = new Func(2, "조범수", "프론트앤드 개발자"); //인스턴스 생성

    info1.result();
    info2.result();

}
