04.switch문 

{
  let num = 100;

  switch( num ){
    case 100 :
      document.write("*** 04. switch문 ***");
      document.write("조건문이 실행되었습니다.( num === 100)");
    break;
    case 200 :
      document.write("*** 04. switch문 ***");
      document.write("조건문이 실행되었습니다.( num === 200)");
    break;
    case 300 :
      document.write("*** 04. switch문 ***");
      document.write("조건문이 실행되었습니다.( num === 300)");
    break;
    default :
      document.write("*** 04. switch문 ***");
      document.write("조건문이 실행되었습니다.(num == 값이 없음)");
    break;
  }
}

07. while문 

{
  document.write("*** 07. while문 ***");
  
  let num = 1;
  while( num <= 10 ){
    document.write(num+" 반복문이 실행되었습니다.");
    num++;
  }
}

08. do while문

{
  document.write("*** 08. do while문 ***");
  
  let num = 1;
  do {
    document.write(num+ "반복문이 실행되었습니다.");
    num++;
   } while (num <= 10);
}

10. 중첩 for문

{
  document.write("*** 10. 중첩 for문 ***");
  for(var i=1; i<=5; i++){
    document.write(i+". 아이 반복문이 실행되었습니다.");
    for(var j=1; j<=5; j++){
      document.write(j+". 제이 반복문이 실행되었습니다.");
    }
  }
}

11. forEach문

{
  document.write("*** 11. forEach() ***");

  const num = [100, 200, 300, 400 , 500];
  
  num.forEach((element ,index, array)=>{
    document.write(element);
    document.write(index);
    document.write(array);
  });
}

12. for in문

{

}

13. for in문

{
  document.write("*** 13. for in문 ***");

  const num = [100, 200, 300, 400, 500];

  for(let i in num){
    document.write(num[i]+". 반복문이 실행되었습니다.");
  }
  
  const obj = {
    a: 100,
    b: 200,
    c: 300,
    d: 400,
    e: 500
  };

  for( let key in obj ){
    document.write(key);
    document.write(obj[key]);
  }
}

14. break문

{
  document.write("*** 14. break문 ***");
  let num = 1;
  while( num < 20 ){
    document.write(num);
    if( num === 10 ){
      break;
    }
    num++;
  }

  for(let i=1; i<=20; i++){      
    document.write(i);
    if( i === 10 ){
      break;
    }
  }
}

15. continue문

{
  document.write("*** 15. continue문 ***");

  let num = 0;
  while( true ){
    num++;
    if( num == 3 ){
      continue;
    }
    if( num > 19 ){
      break;
    }
    document.write(num);
  }
}
