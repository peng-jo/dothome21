<script>>      
    $(".paraNav ul li").click(function(e){
        e.preventDefault();
        let target = $(this);
        let index = target.index();
        let section = $(".content-item").eq(index);
        let offset = section.offset().top;
        $("html, body").animate({scrollTop:offset}, 600, "easeInOutCirc");
    })

    let num = 0;
    let lastScrollTop = 0;
    let nowScrollTop  = false;
    $(window).scroll(function(){
        nowScrollTop = true;
        setTimeout( ()=>{
            if(nowScrollTop){
                nowScrollTop = false;
                hasScroll();
            }
        },1000)
        
    });

    function hasScroll(){
        let scrollTop = $(this).scrollTop();
        $('.paraScroll span').text(scrollTop);
        if( scrollTop > lastScrollTop ){
            $(".paraNav").addClass("hide");
        } else {
            $(".paraNav").removeClass("hide");
        }
        // console.log(scrollTop);
        // console.log(lastScrollTop);
        $(".content-item").each( ( index, obj )=>{
            if( scrollTop >= $(obj).offset().top - 2 ){
                $(".paraNav ul li").eq(index).addClass("active").siblings().removeClass("active");
            }
        });

        lastScrollTop = scrollTop;
    }
</script>
