<style>
    .paraNav {
        top: 20px;
        transition: top 0.4 ease;
    }
    .paraNav li a {
        padding: 5px;
        width: 30px;
        height: 30px;
    }
    .paraNav li.active a {
        border-radius: 30px;
    }
    .content-item .desc span {
        opacity: 0;
        display: inline-block;
        min-width: 1vw;
        transform: translateY(100px) rotate(90deg) scale(2);
    }
</style>
<script>
    $(".content-item .desc").each(function(){
        let text = $(this).text();
        let split = text.split('').join("</span><span aria-hidden='true'>");
        split = "<span aria-hidden='true'>" + split + "</span>";
        $(this).html(split).attr("aria-label",text);
    })

    $(window).scroll(function(){

        // GSAP
        $(".content-item").each(function(index){
            if( $(window).scrollTop() >= $(this).offset().top ){
                gsap.to( $(this).find(".desc span"), { 
                    duration: .5,
                    opacity:1,
                    stagger: 0.05,
                    y: 0,
                    ease: "power3.out",
                    rotation: 0,
                    scale: 1
                })
            //메뉴
            $(".paraNav ul li").eq(index).addClass("active").siblings().removeClass("active");
            }
            
        })
        $(".paraScroll span").text(parseInt($(window).scrollTop()));
    });
</script>
<style>>
    .paraNav {

<script>
    //다중 글씨 쪼개기
    document.querySelectorAll(".split").forEach( el => {
        let text = el.innerText;
        let split = text.split('').join("</span><span aria-hidden='true'>");
            split = "<span aria-hidden='true'>" + split + "</span>";
        el.innerHTML = split;
        el.setAttribute("aria-label", split)
    });

    //글씨 쪼개기
    window.addEventListener("scroll", ()=>{
        let scrollTop = window.pageYOffset || document.documentElement.scrollTop || window.scrollY;
    
        document.querySelectorAll(".content-item").forEach((el,index) => {
            if ( scrollTop >= el.offsetTop - window.innerHeight / 2) {
                gsap.to(el.querySelectorAll(".split span"), {
                    duration: .5,
                    opacity: 1,
                    stagger: 0.05,
                    y: 0,
                    ease: "power3.out",
                    rotation: 0,
                    scale: 1
                });
                //메뉴
                document.querySelectorAll(".paraNav ul li").forEach( li => {
                    li.classList.remove("active");
                })
                document.querySelector(".paraNav ul li:nth-child("+(index+1)+")").classList.add("active");
            }
        })

        //출력
        document.querySelector(".paraScroll span").innerText = parseInt(scrollTop);
    });
</script>
