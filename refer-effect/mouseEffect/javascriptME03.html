<script>>
    const li = document.querySelectorAll(".info.left.list li");
    const cursor = document.querySelector(".cursor");
    li.forEach((el, index)=>{
        el.addEventListener("click",()=>{
            li.forEach( element =>{
                element.classList.remove("active");
            })
            const text = el.textContent || el.innerText;
            cursor.style.mixBlendMode = text;
            el.classList.add("active");
        });
    });
    
    const w = document.querySelector(".cursor").clientWidth / 2;
    const h = document.querySelector(".cursor").clientHeight / 2;
    
    //움직임 효과
    document.addEventListener("mousemove", e =>{
        gsap.to(".cursor", { duration: .5, left:e.pageX - 5, top: e.pageY - 5});
    });
    
    //오버 효과
    document.querySelectorAll(".mouseCont em").forEach( em => {
        em.addEventListener("mouseenter", ()=> {
            document.querySelector(".cursor").classList.add("active");
        });
        em.addEventListener("mouseleave", ()=> {
            document.querySelector(".cursor").classList.remove("active");
        });
    });
</script>

<script>
    const w = $(".cursor").width() / 2;
    const h = $(".cursor").height() / 2;
    
    //움직임 효과
    $(window).mousemove(function(e){
        gsap.to(".cursor", {duration: 0.4, left: e.pageX - w, top: e.pageY - h});
    });
    
    //오버 효과
    $(".mouseCont em").hover(function(){
        $(".cursor").addClass("active");
    },function(){
        $(".cursor").removeClass("active");
    });
    
    const $li = $(".info.left.list li");
    //출력용
    $li.click(function(){
      $li.removeClass("active");
    
        const text = $(this).addClass("active").text();
        $(".cursor").css("mix-blend-mode", text);
    });
</script>
