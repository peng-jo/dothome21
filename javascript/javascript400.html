01. 1부터 100까지 출력하기

{
    document.write("*** . 1부터 100까지 출력 ***");
    for( let i=1; i<=100; i++ ){
        document.write(i);
    }
}

02. 1부터 100까지 출력하기(짝수만 출력)

{
    document.write("*** . 1부터 100까지 짝수만 출력 ***");
    for( let i=1; i<=100; i++ ){
        if( i%2 == 0){
            document.write(i);
        }
    }
}

03. 1부터 100까지 출력하기(5의 배수만 출력)

{
    document.write("*** . 1부터 100까지 5의배수만 출력 ***");
    for( let i=1; i<=100; i++ ){
        if( i%5 == 0){
            document.write(i);
        }
    }
}

04. 1부터 100까지 출력하기(짝수는 빨간색, 홀수는 파란색 출력)

{
    document.write("*** . 1부터 100까지 짝수는빨간색, 홀수는 파란색 출력 ***");
    for( let i=1; i<=100; i++ ){
        if( i%2 === 0 ){
            document.write("<span style='color:red'>"+i+"</span>");
        }else if(i%2 === 1 ){
            document.write("<span style='color:red'>"+i+"</span>");
        }
    }
}

05. 1부터 100까지 합출력하기

{    
    document.write("*** 05 .1부터 100까지 합출력 ***");

    let sum = 0;
    for( let i=1; i<=100; i++ ){
        sum += i;
    }
    document.write(sum)
}

06. 구구단 출력하기

{
    document.write("*** 100. 구구단 만들기 ***");

    /*
        1~9 for(var i=1; i<=9; i++)
        //num

        2*1=2;  3*1=3;
        2*2=4;  3*2=6;
        2*3=6;  3*3=9;
        2*4=8;  3*4=12;
        ....         ....
        2*9=18; 9*1=9;
                    9*2=18;
                    9*9=81;
    */

    for(var i=2; i<=9; i++){
        for(var j=1; j<=9; j++){
            var sum = i * j;
            document.write(i + " * " +j+" = "+sum);
        }
    }
}

07. 테이블 만들기1

{
    document.write("*** 101. 테이블 만들기 ***");
    let table = "<table>";
    let num = 1;
    for( let i=1; i<=10; i++){
      table += "<tr>";
      for( let j=1; j<=10; j++){
        table += "<td>"+ num +"</td>";
        num ++;
      }
    }
    table += "</table>";
    document.write(table)
}

08. 테이블 만들기2

{
    document.write("*** 2. 테이블 만들기2 ***");
    let table = "<table>";
    let num = 100;
    for( let i=1; i<=10; i++){
      table += "<tr>";
      for( let j=1; j<=10; j++){
        table += "<td>"+ num +"</td>";
        num --;
      }
    }
    table += "</table>";
}

09. 테이블 만들기3 - 짝수만

{
    document.write("*** 102. 테이블 만들기3 ***");
    
    let table = "<table>";
    let num = 2;
    for( let i=1; i<=10; i++){
      table += "<tr>";
      for( let j=1; j<=10; j++){
        if( num%2===0){
          table += "<td>"+ num +"</td>";
        }
        num += 1;
      }
    }
    table += "</table>";
    document.write(table);
