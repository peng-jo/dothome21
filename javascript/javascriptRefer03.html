19. 콜백 함수 : 동기/비동기 

{
    //01
    document.write("*** 19. 콜백 함수 : 동기/비동기 *** ");

    function funcA(){
        document.write("funcA가 실행되었습니다.");

    }

    function funcB(){
        document.write("funcB가 실행되었습니다.");
    }
    funcA();
    funcB();

    //02
    function funcC(){
        setTimeout(function(){
            console.log("funC가 실행되었습니다.");
        },1000);

    }

    function funcD(){
        console.log("funcD가 실행되었습니다.");
    }
    funcC();
    funcD();

    //03
    function funcE(callback){
        setTimeout(function(){
            console.log("funE가 실행되었습니다.");
            callback()
        },1000);

    }
    function funcF(){
        console.log("funF가 실행되었습니다.");
    }
    funcE(funcF);
}
